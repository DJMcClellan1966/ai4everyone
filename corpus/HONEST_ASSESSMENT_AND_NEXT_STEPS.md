```python
# simple_ml_helper.py
from ml_toolbox import MLToolbox
import numpy as np

def train_model(X, y, task_type='classification'):
    """Simple ML helper - auto-selects best model"""
    toolbox = MLToolbox()
    result = toolbox.fit(X, y, task_type=task_type, preprocess=False)
    return result['model'], result['metrics']

# Use it
X = np.random.rand(100, 5)
y = np.random.randint(0, 2, 100)
model, metrics = train_model(X, y)
print(f"Accuracy: {metrics.get('accuracy', 0):.2f}")
```
